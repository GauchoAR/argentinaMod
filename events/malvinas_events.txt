namespace = malvinas

# An Offer for malvinas
malvinas.1 = {
	type = country_event
	placement = ROOT

	title = malvinas.1.t
	desc = malvinas.1.d
	flavor = malvinas.1.f

	duration = 3

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	trigger = {
	}

	immediate = {
		c:ARG = {
			save_scope_as = arg_scope
		}
		s:STATE_ISLAS_MALVINAS.region_state:ARG = {
			save_scope_as = malvinas_scopeMalvinas
		}
	}

	option = {
		name = malvinas.1.a
		default_option = yes
		add_modifier = {
			name = malvinas_gbr_big_money
			months = 12
			is_decaying = no
		}
		c:ARG = {
			trigger_event = { id = malvinas.2 days = 0 popup = yes }
		}
		show_as_tooltip = {
			s:STATE_ISLAS_MALVINAS.region_state:ARG = {
				set_state_owner = c:ARG
			}
			s:STATE_ISLAS_GEORGIAS.region_state:ARG = {
				set_state_owner = c:ARG
			}
			change_relations = {
				country = c:ARG
				value = 10
			}
		}
		ai_chance = {
			base = 1
		}
	}
	option = {
		name = malvinas.1.b
		c:ARG = {
			trigger_event = { id = malvinas.3 days = 0 popup = yes }
		}
		show_as_tooltip = {
			change_relations = {
				country = c:ARG
				value = -20
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
