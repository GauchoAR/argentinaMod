namespace = vientres

vientres.1 = { 
	type = country_event
	hidden = yes

	trigger = {
	}
	
	immediate = {
    every_scope_pop = {
        if = {
			limit = {
				is_pop_type = slaves
				}
			change_pop_religion = { target = rel:slave value = 0.2 }
		}
    }
    every_scope_pop = {
        if = {
			limit = {
				has_pop_religion = slave
				}
			change_poptype = pop_type:laborers
		}	
    }
    every_scope_pop = {
        if = {
			limit = {
				has_pop_religion = slave
				}
            change_pop_religion = { target = rel:catholic value = 1.0 }
		}	
    }
    }
}
