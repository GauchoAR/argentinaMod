arg_purchase_malvinas = {
	is_shown = {
		exists = c:ARG
		exists = c:GBR
		c:ARG = ROOT
		is_subject = no
		NOT = {
			has_variable = malvinas_purchase_var
		}
		c:GBR = {
			owns_entire_state_region = STATE_ISLAS_MALVINAS
			owns_entire_state_region = STATE_ISLAS_GEORGIAS
			is_subject = no
		}
	}

	possible = {
		in_default = no
		taking_loans = no
		is_at_war = no
		c:GBR = {
			relations:root >= 80
			is_at_war = no
			
		}

		}
	

	when_taken = {
		set_variable = {
			name = malvinas_purchase_var
			value = yes
		}
		trigger_event = {
			id = malvinas.4
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 0	
		modifier = {
			trigger = { gold_reserves > 1000000 }
			add = 10
		}		
		modifier = {
			trigger = { 
				gold_reserves > 0
				weekly_net_fixed_income > 50000
			}
			add = 10
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}
